1. prettier-plugin-sort-imports自定义模块导入顺序#

可以自定义模块导入顺序

npm install --save-dev @trivago/prettier-plugin-sort-imports
module.exports = {
"printWidth": 80,
"tabWidth": 4,
"trailingComma": "all",
"singleQuote": true,
"jsxBracketSameLine": true,
"semi": true,
"importOrder": ["^@core/(.*)$", "^@server/(.*)$", "^@ui/(.*)$", "^[./]"],
"importOrderSeparation": true,
}
